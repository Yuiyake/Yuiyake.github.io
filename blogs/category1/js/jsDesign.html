    # 发布-订阅模式

    • 发布-订阅模式：三个模块，发布者，订阅者，调度中心。
    • 调度中心相当于办事大厅，发布者相当于杂志负责人，来中心注册事件，而订阅者相当于用户，在中心订阅量 这份杂志。
    • 每当发布者发布了一期杂志，订阅中心就会通知订阅者来拿，这就是发布订阅模式。
    class Event {
    constructor() { };
        handlers = {};
        // 添加事件
        addEventListener(type, handler) {
            if(!(type in this.handlers)) this.handlers[type] = [];
            this.handlers[type].push(handler);
        }
        // 执行事件（调度中心）
        dispathEvent(type, ...params){
            if(!(type in this.handlers)) return new Error('未注册事件！');
            this.handlers[type].forEach(handler => {
                handler(...params);
            })
        }
        // 移除事件
        removeEvent(type, handler){
            if(!(type in this.handlers)) return new Error('无效事件！');
            if(!handler) delete this.handlers[type];
            else {
                const index = this.handlers[type].findIndex(ele => ele === handler);
                if(index === undefined) return new Error('未找到该绑定事件');
                this.handlers[type].splice(index, 1);
                if(this.handlers[type].length === 0) delete this.handlers[type];
            }
        }
}
// 测试实例
var event = new Event();
function load() {
    console.log("load", load);
}
function load2(params){
    console.log('load2', params);
}
// 添加事件load，load2，调度中心调度事件，然后删除load2，删除所有load
event.addEventListener('load', load);
event.addEventListener('load', load2);
event.dispathEvent('load', 'load被触发');
event.removeEvent('load', load2);
event.removeEvent('load');

    # 观察者模式

    let obs = function observer(oldVal, newVal) {
    console.info('name属性从' + oldVal + '变为' + newVal)
}
class TargetObj {
    constructor(name) {
        this.name = name
    }
    set name(val){
        obs(name, val)
        name = val;
    }
}
let targetObj = new TargetObj('John')
targetObj.name = 'Mike'
console.log(targetObj)  //name属性从John变为Mike
